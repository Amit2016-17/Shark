46 /// \brief An optimizer that optimizes general objective functions
47 ///
48 /// After construction and configurationg the optimizer, init() is called with the objective function
49 /// to be used. After that step() can be called until the required solution is found. The solution can be queried
50 /// using solution(). The type of the solution depends on the optimisation problem at hand.
51 /// It is allowed to add constrains on the features the objective function needs to offer
52 ///
53 /// These are:
54 /// - REQUIRES_VALUE: The function is evaluated to use the optimizer and
55 /// the HAS_VALUE-flag must be set
56 /// - REQUIRES_FIRST_DERIVATIVE: The first derivative needs to be evaluated and
57 /// - HAS_FIRST_DERIVATIVE must be set
58 /// - REQUIRES_SECOND_DERIVATIVE: The second derivative needs to be evaluated and
59 /// - HAS_SECOND_DERIVATIVE must be set
60 /// - CAN_SOLVE_CONSTRAINED: The optimizer can solve functions which are constrained and
61 /// where the IS_CONSTRAINED_FEATURE is set.
62 /// - REQUIRES_CLOSEST_FEASIBLE: If the function is constrained, it must offer a way to
63 /// construct the closest feasible point and
64 /// - CAN_PROVIDE_CLOSEST_FEASIBLE must be set
65 ///
66 /// Also when init() is called as offered by the AbstractOptimizer interface, the function
67 /// is required to have the CAN_PROPOSE_STARTING_POINT flag.
68 ///
69 /// \tparam PointType The type of search space the optimizer works upon.
70 /// \tparam ResultT The objective space the optimizer works upon.
71 /// \tparam SolutionTypeT The type of the final solution.
72 /// \ingroup optimizers
73 template <typename PointType, typename ResultT, typename SolutionTypeT>
74 class AbstractOptimizer : public INameable, public ISerializable {
75 public:
76  typedef PointType SearchPointType;
77  typedef ResultT ResultType;
78  typedef SolutionTypeT SolutionType;
79  typedef AbstractObjectiveFunction<PointType,ResultType> ObjectiveFunctionType;
80 
81  /// \brief Models features that the optimizer requires from the objective function.
82  /// \sa AbstractObjectiveFunction
83  enum Feature {
84  REQUIRES_VALUE = 1,
85  REQUIRES_FIRST_DERIVATIVE = 2,
86  REQUIRES_SECOND_DERIVATIVE = 4,
87  CAN_SOLVE_CONSTRAINED = 8,
88  REQUIRES_CLOSEST_FEASIBLE = 16
89  };
90 
91  SHARK_FEATURE_INTERFACE;
92 
93  bool requiresValue()const{
94  return features()& REQUIRES_VALUE;
95  }
96 
97  bool requiresFirstDerivative()const{
98  return features()& REQUIRES_FIRST_DERIVATIVE;
99  }
100  bool requiresSecondDerivative()const{
101  return features()& REQUIRES_SECOND_DERIVATIVE;
102  }
103  bool canSolveConstrained()const{
104  return features()& CAN_SOLVE_CONSTRAINED;
105  }
106  bool requiresClosestFeasible()const{
107  return features()& REQUIRES_CLOSEST_FEASIBLE;
108  }
109 
110  virtual ~AbstractOptimizer() {}
111 
112  /// \brief Returns the number of points this method requires for initialisation
113  ///
114  /// The number of points supplied is allowed to be smaller, however in this case
115  /// the optimizer will resort to techniques generating additional points if needed.
116  virtual std::size_t numInitPoints() const = 0;
117 
118  /// \brief Initialize the optimizer for the supplied objective function.
119  ///
120  /// Be aware that function.init() has to be called before calling this function!
121  /// This function will initialize the algorithm with a number of points proposed
122  /// by the function. Note that this must fail if the function can not propose starting point(s).
123  ///
124  /// \param [in] function The objective function to initialize for.
125  virtual void init( ObjectiveFunctionType const& function ){
126  SHARK_RUNTIME_CHECK(function.canProposeStartingPoint(), "Objective function does not propose a starting point");
127  std::vector<SearchPointType> initPoints(numInitPoints());
128  for(SearchPointType& point: initPoints){
129  point = function.proposeStartingPoint();
130  }
131  init(function,initPoints);
132  }
133 
134 
135  /// \brief Initialize the optimizer for the supplied objective function using a set of initialisation points
136  ///
137  /// Most single objective algorithms only require a single point. However multi-objective algorithms
138  /// need a set of initialisation points. The number of points required should be numInitPoints().
139  /// Otherwise the algorithm might use heuristics to generate additional points if needed.
140  ///
141  /// Be aware that function.init() has to be called before calling this function!
142  /// \param [in] function The objective function to initialize for.
143  /// \param [in] initPoints points used for initialisation. Should be at least numInitPoints().
144  virtual void init( ObjectiveFunctionType const& function, std::vector<SearchPointType> const& initPoints ) = 0;
145 
146  /// \brief Carry out one step of the optimizer for the supplied objective function.
147  /// \param [in] function The objective function to initialize for.
148  virtual void step( ObjectiveFunctionType const& function ) = 0;
149 
150  /// \brief Accesses the best solution obtained so far.
151  /// \return An immutable reference to the best solution obtained so far.
152  virtual SolutionType const& solution() const = 0; //mt_hint: try accessing this thing via solution().point and solution().value..
153 
154 protected:
155  /// \brief Convenience function that checks whether the features of the supplied objective function match with the required features of the optimizer.
156  /// \param [in] objectiveFunction The function to match with.
157  /// \throws shark::Exception
158  void checkFeatures (ObjectiveFunctionType const& objectiveFunction){
159  //test whether the function can be evaluated
160  SHARK_RUNTIME_CHECK(!requiresValue() || objectiveFunction.hasValue(), name()+" Requires value of objective function");
161  //test first derivative
162  SHARK_RUNTIME_CHECK(!requiresFirstDerivative() || objectiveFunction.hasFirstDerivative(), name()+" Requires first derivative of objective function");
163  //test second derivative
164  SHARK_RUNTIME_CHECK(!requiresSecondDerivative() || objectiveFunction.hasSecondDerivative(), name()+" Requires second derivative of objective function");
165  //test for constraints
166  if(objectiveFunction.isConstrained()){
167  SHARK_RUNTIME_CHECK(canSolveConstrained(), name()+" Can not solve constrained problems");
168  SHARK_RUNTIME_CHECK(
169  !requiresClosestFeasible() || objectiveFunction.canProvideClosestFeasible(),
170  name()+" Requires closest feasible solution for solving constrained problems"
171  );
172  }
173  }
174 };

176 }
