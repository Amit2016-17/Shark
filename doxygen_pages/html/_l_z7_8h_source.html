87  ResultType eval( const SearchPointType & x ) const {
88  m_evaluationCounter++;
89 
90  ResultType value( 2, 0 );
91 
92  std::size_t counter1 = 0, counter2 = 0;
93  for( std::size_t i = 1; i < x.size(); i++ ) {
94  double y = x(i) - ::pow( x(0), 0.5*(1.0 + 3*(i-1)/(x.size()-1) ) );
95  if( i % 2 == 0 ) {
96  counter2++;
97  value[1] += 4*sqr( y ) - ::cos( 8*y*M_PI) + 1.;
98  } else {
99  counter1++;
100  value[0] += 4*sqr( y ) - ::cos( 8*y*M_PI) + 1.;
101  }
102  }
103 
104  value[0] *= 2./counter1;
105  value[0] += x( 0 );
106 
107  value[1] *= 2./counter2;
108  value[1] += 1 - std::sqrt( x( 0 ) );
109 
110  return value;
111  }
112 private:
