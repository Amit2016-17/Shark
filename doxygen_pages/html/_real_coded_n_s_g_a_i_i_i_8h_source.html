42 /// \brief Implements the NSGA-III
43 ///
44 /// The NSGAIII works similar to the NSGAII except that the crowding distance is replaced by its own,
45 /// reference point based indicator.
46 ///
47 /// Please see the following papers for further reference:
48 /// Deb, Kalyanmoy, and Himanshu Jain.
49 /// "An evolutionary many-objective optimization algorithm using
50 /// reference-point-based nondominated sorting approach,
51 /// part I: Solving problems with box constraints."
52 /// IEEE Trans. Evolutionary Computation 18.4 (2014): 577-601.
53 /// \ingroup multidirect
54 class RealCodedNSGAIII : public IndicatorBasedRealCodedNSGAII<NSGA3Indicator>{
55  typedef IndicatorBasedRealCodedNSGAII<NSGA3Indicator> base;
56 public:
57 
58  /// \brief Default c'tor.
59  RealCodedNSGAIII(random::rng_type& rng = random::globalRng)
60  : base(rng){}
61 
62  std::string name() const {
63  return "RealCodedNSGAIII";
64  }
65 protected:
66  void doInit(
67  std::vector<SearchPointType> const& initialSearchPoints,
68  std::vector<ResultType> const& functionValues,
69  RealVector const& lowerBounds,
70  RealVector const& upperBounds,
71  std::size_t mu,
72  double nm,
73  double nc,
74  double crossover_prob,
75  std::vector<Preference> const & indicatorPreferences = std::vector<Preference>()
76  ){
77  // Do the regular initialization.
78  base::doInit(
79  initialSearchPoints,
80  functionValues,
81  lowerBounds,
82  upperBounds,
83  mu,
84  nm,
85  nc,
86  crossover_prob);

88  // are set.
89  indicator().init(functionValues.front().size(), mu, *mpe_rng,
90  indicatorPreferences);
91  }
92 };
