149  /// \brief Initializes the algorithm for the supplied objective function.
150  ///
151  /// \param [in] function The objective function.
152  /// \param [in] initialSearchPoints A set of intiial search points.
153  void init(
154  ObjectiveFunctionType const& function,
155  std::vector<SearchPointType> const& initialSearchPoints
156  ){
157  checkFeatures(function);
158  std::vector<RealVector> values(initialSearchPoints.size());
159  for(std::size_t i = 0; i != initialSearchPoints.size(); ++i){
160  SHARK_RUNTIME_CHECK(function.isFeasible(initialSearchPoints[i]),"Starting point(s) not feasible");
161  values[i] = function.eval(initialSearchPoints[i]);
162  }
163 
164  std::size_t dim = function.numberOfVariables();
165  RealVector lowerBounds(dim, -1E20);
166  RealVector upperBounds(dim, 1E20);
167  if (function.hasConstraintHandler() && function.getConstraintHandler().isBoxConstrained()) {
168  typedef BoxConstraintHandler<SearchPointType> ConstraintHandler;
169  ConstraintHandler const& handler = static_cast<ConstraintHandler const&>(function.getConstraintHandler());
170 
171  lowerBounds = handler.lower();
172  upperBounds = handler.upper();
173  } else{
174  SHARK_RUNTIME_CHECK(
175  !function.isConstrained(), "Algorithm does only allow box constraints"
176  );
177  }
178 
179  doInit(initialSearchPoints,values,lowerBounds, upperBounds, mu(), nm(), nc(), crossoverProbability());
180  }

182  /// \brief Executes one iteration of the algorithm.
183  ///
184  ///\param [in] function The function to iterate upon.
185  void step( ObjectiveFunctionType const& function ) {
186  std::vector<IndividualType> offspring = generateOffspring();
187  PenalizingEvaluator penalizingEvaluator;
188  penalizingEvaluator( function, offspring.begin(), offspring.end() );
189  updatePopulation(offspring);
190  }

195  void doInit(
196  std::vector<SearchPointType> const& initialSearchPoints,
197  std::vector<ResultType> const& functionValues,
198  RealVector const& lowerBounds,
199  RealVector const& upperBounds,
200  std::size_t mu,
201  double nm,
202  double nc,
203  double crossover_prob
204  ){
205  SIZE_CHECK(initialSearchPoints.size() > 0);
206  m_mu = mu;
207  m_mutation.m_nm = nm;
208  m_crossover.m_nc = nc;
209  m_crossoverProbability = crossover_prob;
210  m_best.resize( mu );
211  m_parents.resize( mu );
212  //if the number of supplied points is smaller than mu, fill everything in
213  std::size_t numPoints = 0;
214  if(initialSearchPoints.size()<=mu){
215  numPoints = initialSearchPoints.size();
216  for(std::size_t i = 0; i != numPoints; ++i){
217  m_parents[i].searchPoint() = initialSearchPoints[i];
218  m_parents[i].penalizedFitness() = functionValues[i];
219  m_parents[i].unpenalizedFitness() = functionValues[i];
220  }
221  }
222  //copy points randomly
223  for(std::size_t i = numPoints; i != mu; ++i){
224  std::size_t index = random::discrete(*mpe_rng, std::size_t(0),initialSearchPoints.size()-1);
225  m_parents[i].searchPoint() = initialSearchPoints[index];
226  m_parents[i].penalizedFitness() = functionValues[index];
227  m_parents[i].unpenalizedFitness() = functionValues[index];
228  }
229  //create initial mu best points
230  for(std::size_t i = 0; i != mu; ++i){
231  m_best[i].point = m_parents[i].searchPoint();
232  m_best[i].value = m_parents[i].unpenalizedFitness();
233  }
234  m_crossover.init(lowerBounds,upperBounds);
235  m_mutation.init(lowerBounds,upperBounds);
236  indicator().init(functionValues.front().size(),mu,*mpe_rng);
237  m_selection( m_parents, mu );
238 
239 
240  }

242  std::vector<IndividualType> generateOffspring()const{
243  TournamentSelection< IndividualType::RankOrdering > selection;
244  std::vector<IndividualType> offspring(mu());
245  selection(
246  *mpe_rng,
247  m_parents.begin(),
248  m_parents.end(),
249  offspring.begin(),
250  offspring.end()
251  );
252 
253  for( std::size_t i = 0; i < mu()-1; i+=2 ) {
254  if( random::coinToss(*mpe_rng, m_crossoverProbability ) ) {
255  m_crossover(*mpe_rng,offspring[i], offspring[i +1] );
256  }
257  }
258  for( std::size_t i = 0; i < mu(); i++ ) {
259  m_mutation(*mpe_rng, offspring[i] );
260  }
261  return offspring;
262  }
263 
264  void updatePopulation( std::vector<IndividualType> const& offspringVec) {
265  m_parents.insert(m_parents.end(),offspringVec.begin(),offspringVec.end());
266  m_selection( m_parents, mu());
267 
268  //partition the selected individuals to the front and remove the unselected ones
269  std::partition(m_parents.begin(), m_parents.end(), [](IndividualType const& ind){return ind.selected();});
270  m_parents.erase(m_parents.begin()+mu(),m_parents.end());
271 
272  //update solution set
273  for (std::size_t i = 0; i < mu(); i++) {
274  noalias(m_best[i].point) = m_parents[i].searchPoint();
275  m_best[i].value = m_parents[i].unpenalizedFitness();
276  }
277  }
